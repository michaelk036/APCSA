public static int numberOfLeapYears (int year1, int year2) {
    int yearOne = year1;
    int yearTwo = year2; 
    int counter = 0;

    for (int i = yearOne; i <= yearTwo; i++) { //make sure upperbound_lowerbound is correct
        if (isLeapYear(i)) {
            counter++;
        }
    }
    return counter;
}

public static int dayOfWeek(int month, int day, int year) {
    int firstDay = firstDayofYear(year);  
    int day = dayOfYear(month, day, year);

    int output = ((firstDay + day)-1)%7;
    return output;
}

- isLeapYear(i) not isLeapYear(year)
- -1 because a month starts at 1 not at 0
