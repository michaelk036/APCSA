public boolean simulate() {
    int pos;

    for (int i = 0; i < maxHops; i++) {
        pos += hopDistance();

        if (pos > goalDistance) {
            return true;
        }
        else if (pos < 0) {
            return false;
        }
    }
    return false;
}

public double runSimulation (int num) {
    int success = 0;

    if (simulate()) {
        success++;
    }

    double output = (double) success/num;
    return output;
}
