public boolean simulate() {
    int pos = 0;

    for (int i = 0; i < maxHops; i++) {
        pos += hopDistance();

        if (pos >= goalDistance) {
            return true;
        }
        else if (pos < 0) {
            return false; 
        }
    }
    return false;
}

public double runSimulations (int num) {
    int success = 0;

    for (int i = 0; i < num; i++) {
        if (simulate()) {
            success++;
        }
    }

    double output = (double) success/num;
    return output;
}

notes

- return serves as a break
- runSimulations not runSimulation
- pay attention to <= vs <
- define before +=

for (int i = 0; i < num; i++) could also be (int i = 1; i <= num; i++) - doesn't matter
