public String[] getWords() {
    ArrayList<Integer> blanks = getBlankPositions();
    String array[] = new String[countWords()];

    int start = 0;

    for (int i = 0; i < blanks.size(); i++) {
        int end = blanks.get(i);
        array[i] = sentence.substring(start, end);
        start = end + 1;
    }

    array[array.length] = sentence.substring(start);
    return array;
}
