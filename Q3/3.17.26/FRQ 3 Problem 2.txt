public static ArrayList<Integer> findRPeaks(double[] v) {
    ArrayList<Integer> peaks = new ArrayList<>();

    for (i >= v[DELTA]; i < v[v.length-1-DELTA]; i++) {
        if (isRPeak(v, DELTA)) {
            peaks.add(DELTA);
        } 
        else {
            DELTA++;
        }
    }
    return peaks;
}
